cmake_minimum_required(VERSION 3.10)

# Set the project name and version
project(newMiner VERSION 1.0)

# Specify the C++ standard
set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_STANDARD_REQUIRED True)

# Include the directory containing the shared CMakeLists.txt
add_subdirectory(../path/to/shared CMakeFiles/SharedFile)

# Add the miner executable
add_executable(newMiner newMiner.cpp)

# Link the static library to the miner executable
target_link_libraries(newMiner SharedFile)
